84 inline void ansicolor_sink::log(const details::log_msg& msg)
85 {
86  // Wrap the originally formatted message in color codes
87  const std::string& prefix = colors_[msg.level];
88  const std::string& s = msg.formatted.str();
89  const std::string& suffix = reset;
90  details::log_msg m;

92  m.logger_name = msg.logger_name;
93  m.time = msg.time;
94  m.thread_id = msg.thread_id;
95  m.formatted << prefix << s << suffix;
